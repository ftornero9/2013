          "Valido hasta"       := DT2DATE(Cert.NotAfter);
          "Huella digital"     := COPYSTR(Cert.Thumbprint, 1, MAXSTRLEN("Huella digital"));

            Password           := Encrypt(Password);
            Encrypted          := TRUE;
          END;

      END;
